import requests
from sqlalchemy.orm import Session
from sqlalchemy.exc import IntegrityError
from app.core.database import SessionLocal
from app.models.book import Book
from app.models.tag import Tag
from app.models.category import Category

OPENLIBRARY_URL = "https://openlibrary.org/search.json"
BATCH_SIZE = 100


def get_cover_url_from_doc(doc):
    """Retourne l'URL de la couverture à partir de cover_i ou isbn"""
    if doc.get("cover_i"):
        return f"https://covers.openlibrary.org/b/id/{doc['cover_i']}-L.jpg"
    isbn_list = doc.get("isbn", [])
    if isbn_list:
        return f"https://covers.openlibrary.org/b/isbn/{isbn_list[0]}-L.jpg"
    return None


def get_or_create(db: Session, model, **kwargs):
    """Récupère l'objet s'il existe sinon le crée"""
    instance = db.query(model).filter_by(**kwargs).first()
    if instance:
        return instance
    instance = model(**kwargs)
    db.add(instance)
    try:
        db.commit()  # commit pour récupérer l'id
        db.refresh(instance)
    except IntegrityError:
        db.rollback()
        instance = db.query(model).filter_by(**kwargs).first()
    return instance


def import_books(query: str, pages: int = 5):
    db = SessionLocal()
    imported = 0

    try:
        for page in range(1, pages + 1):
            print(f"📦 Page {page}")
            response = requests.get(
                OPENLIBRARY_URL,
                params={"q": query, "page": page, "limit": BATCH_SIZE},
                timeout=15
            )
            response.raise_for_status()
            docs = response.json().get("docs", [])

            for doc in docs:
                isbn_list = doc.get("isbn", [])
                isbn = isbn_list[0] if isbn_list else None

                # Éviter doublons sur ISBN
                if isbn and db.query(Book).filter(Book.isbn == isbn).first():
                    continue

                title = doc.get("title")
                if not title:
                    continue

                author_list = doc.get("author_name", [])
                author = author_list[0] if author_list else None

                # Description : first_sentence ou subtitle
                description = None
                if doc.get("first_sentence"):
                    if isinstance(doc["first_sentence"], dict):
                        description = doc["first_sentence"].get("value")
                    else:
                        description = doc["first_sentence"]
                elif doc.get("subtitle"):
                    description = doc["subtitle"]

                book = Book(
                    title=title,
                    author=author,
                    isbn=isbn,
                    description=description,
                    published_year=doc.get("first_publish_year"),
                    cover_url=get_cover_url_from_doc(doc)
                )

                # Category (premier sujet)
                subjects = doc.get("subject", [])
                if subjects:
                    category_name = subjects[0][:100]  # limite 100 caractères
                    category = get_or_create(db, Category, name=category_name)
                    book.category = category

                    # Tags (max 5)
                    tag_objs = []
                    for tag_name in subjects[:5]:
                        tag = get_or_create(db, Tag, name=tag_name[:100])
                        tag_objs.append(tag)
                    book.tags = tag_objs

                db.add(book)
                imported += 1

            db.commit()
            print(f"✅ {imported} livres importés")

        print("🎉 Import terminé")

    except Exception as e:
        db.rollback()
        print("❌ Erreur :", e)

    finally:
        db.close()


if __name__ == "__main__":
    import_books("science fiction", pages=5)
